1.改变this指向三种方法

1.call
let obj = {
  name: 'evan',
  age: 18
}
function fn(a, b) {
  console.log(this.name,this.age, a, b);
}
fn.call(obj, 1, 2)

调用fn并使用obj中的this

2.apply
let obj = {
  name: 'evan',
  age: 18
}
function fn(a, b) {
  console.log(this.name,this.age, a, b);
}
fn.apply(obj, [1, 2])

调用fn并使用obj中的this

两者的不同在于参数数组的区别,如果参数复杂就用call,如何不复杂就用apply,视情况而定
3.bind

bind和其他两者的区别是返回的是一个改变this指向的函数,而不是直接调用(call和apply返回的是undefined)

let obj = {
  name: 'evan',
  age: 18
}
function fn(a, b) {
  console.log(this.name,this.age, a, b);
}
const a = fn.bind(obj, 1, 2)
a()
